- [ ] web3-react
- [ ] ether.js
- [ ] next.js
- [ ] swr
- [ ] typechain
- [ ] tailwindcss

- [ ] <https://github.com/blacksmith-eth/blacksmith>
- [ ] <https://github.com/wagmi-dev/awesome-wagmi>
